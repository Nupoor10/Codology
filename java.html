Module 4 : File I/O


            // Reading from File
            try {
                BufferedReader reader = new BufferedReader(new FileReader("file.txt"));
                String line;
                while ((line = reader.readLine()) != null) {
                    System.out.println(line);
                }
                reader.close();
            } catch (IOException e) {
                // handle exception
            }

            // Writing to File
